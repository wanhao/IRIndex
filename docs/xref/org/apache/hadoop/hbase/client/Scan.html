42  /**
43   * Used to perform Scan operations.
44   * <p>
45   * All operations are identical to {@link Get} with the exception of
46   * instantiation.  Rather than specifying a single row, an optional startRow
47   * and stopRow may be defined.  If rows are not specified, the Scanner will
48   * iterate over all rows.
49   * <p>
50   * To scan everything for each row, instantiate a Scan object.
51   * <p>
52   * To modify scanner caching for just this scan, use {@link #setCaching(int) setCaching}.
53   * If caching is NOT set, we will use the caching value of the hosting
54   * {@link HTable}.  See {@link HTable#setScannerCaching(int)}.
55   * <p>
56   * To further define the scope of what to get when scanning, perform additional
57   * methods as outlined below.
58   * <p>
59   * To get all columns from specific families, execute {@link #addFamily(byte[]) addFamily}
60   * for each family to retrieve.
61   * <p>
62   * To get specific columns, execute {@link #addColumn(byte[], byte[]) addColumn}
63   * for each column to retrieve.
64   * <p>
65   * To only retrieve columns within a specific range of version timestamps,
66   * execute {@link #setTimeRange(long, long) setTimeRange}.
67   * <p>
68   * To only retrieve columns with a specific timestamp, execute
69   * {@link #setTimeStamp(long) setTimestamp}.
70   * <p>
71   * To limit the number of versions of each column to be returned, execute
72   * {@link #setMaxVersions(int) setMaxVersions}.
73   * <p>
74   * To limit the maximum number of values returned for each call to next(),
75   * execute {@link #setBatch(int) setBatch}.
76   * <p>
77   * To add a filter, execute {@link #setFilter(org.apache.hadoop.hbase.filter.Filter) setFilter}.
78   * <p>
79   * Expert: To explicitly disable server-side block caching for this scan,
80   * execute {@link #setCacheBlocks(boolean)}.
81   */

144   /**
145    * Creates a new instance of this class while copying all values.
146    *
147    * @param scan  The scan instance to copy from.
148    * @throws IOException When copying the values fails.
149    */
150   public Scan(Scan scan) throws IOException {
151     startRow = scan.getStartRow();
152     stopRow  = scan.getStopRow();
153     maxVersions = scan.getMaxVersions();
154     batch = scan.getBatch();
155     caching = scan.getCaching();
156     cacheBlocks = scan.getCacheBlocks();
157     filter = scan.getFilter(); // clone?
158     TimeRange ctr = scan.getTimeRange();
159     tr = new TimeRange(ctr.getMin(), ctr.getMax());
160     Map<byte[], NavigableSet<byte[]>> fams = scan.getFamilyMap();
161     for (Map.Entry<byte[],NavigableSet<byte[]>> entry : fams.entrySet()) {
162       byte [] fam = entry.getKey();
163       NavigableSet<byte[]> cols = entry.getValue();
164       if (cols != null && cols.size() > 0) {
165         for (byte[] col : cols) {
166           addColumn(fam, col);
167         }
168       } else {
169         addFamily(fam);
170       }
171     }
172     for (Map.Entry<String, byte[]> attr : scan.getAttributesMap().entrySet()) {
173       setAttribute(attr.getKey(), attr.getValue());
174     }
175   }

218     NavigableSet<byte []> set = familyMap.get(family);

222     if (qualifier == null) {

224     }
225     set.add(qualifier);
226     familyMap.put(family, set);

228     return this;

466   /**
467    * Set the value indicating whether loading CFs on demand should be allowed (cluster
468    * default is false). On-demand CF loading doesn't load column families until necessary, e.g.
469    * if you filter on one column, the other column family data will be loaded only for the rows
470    * that are included in result, not all rows like in normal case.
471    * With column-specific filters, like SingleColumnValueFilter w/filterIfMissing == true,
472    * this can deliver huge perf gains when there's a cf with lots of data; however, it can
473    * also lead to some inconsistent results, as follows:
474    * - if someone does a concurrent update to both column families in question you may get a row
475    *   that never existed, e.g. for { rowKey = 5, { cat_videos => 1 }, { video => "my cat" } }
476    *   someone puts rowKey 5 with { cat_videos => 0 }, { video => "my dog" }, concurrent scan
477    *   filtering on "cat_videos == 1" can get { rowKey = 5, { cat_videos => 1 },
478    *   { video => "my dog" } }.
479    * - if there's a concurrent split and you have more than 2 column families, some rows may be
480    *   missing some column families.
481    */
482   public void setLoadColumnFamiliesOnDemand(boolean value) {
483     setAttribute(ONDEMAND_ATTR, Bytes.toBytes(value));
484   }

527   /**
528    * Compile the table and column family (i.e. schema) information
529    * into a String. Useful for parsing and aggregation by debugging,
530    * logging, and administration tools.
531    * @return Map
532    */
533   @Override
534   public Map<String, Object> getFingerprint() {
535     Map<String, Object> map = new HashMap<String, Object>();
536     List<String> families = new ArrayList<String>();
537     if(this.familyMap.size() == 0) {
538       map.put("families", "ALL");
539       return map;
540     } else {
541       map.put("families", families);
542     }
543     for (Map.Entry<byte [], NavigableSet<byte[]>> entry :
544         this.familyMap.entrySet()) {
545       families.add(Bytes.toStringBinary(entry.getKey()));
546     }
547     return map;
548   }
549 
550   /**
551    * Compile the details beyond the scope of getFingerprint (row, columns,
552    * timestamps, etc.) into a Map along with the fingerprinted information.
553    * Useful for debugging, logging, and administration tools.
554    * @param maxCols a limit on the number of columns output prior to truncation
555    * @return Map
556    */
557   @Override
558   public Map<String, Object> toMap(int maxCols) {
559     // start with the fingerpring map and build on top of it
560     Map<String, Object> map = getFingerprint();
561     // map from families to column list replaces fingerprint's list of families
562     Map<String, List<String>> familyColumns =
563       new HashMap<String, List<String>>();
564     map.put("families", familyColumns);
565     // add scalar information first
566     map.put("startRow", Bytes.toStringBinary(this.startRow));
567     map.put("stopRow", Bytes.toStringBinary(this.stopRow));
568     map.put("maxVersions", this.maxVersions);
569     map.put("batch", this.batch);
570     map.put("caching", this.caching);
571     map.put("cacheBlocks", this.cacheBlocks);
572     List<Long> timeRange = new ArrayList<Long>();
573     timeRange.add(this.tr.getMin());
574     timeRange.add(this.tr.getMax());
575     map.put("timeRange", timeRange);
576     int colCount = 0;
577     // iterate through affected families and list out up to maxCols columns
578     for (Map.Entry<byte [], NavigableSet<byte[]>> entry :
579       this.familyMap.entrySet()) {
580       List<String> columns = new ArrayList<String>();
581       familyColumns.put(Bytes.toStringBinary(entry.getKey()), columns);
582       if(entry.getValue() == null) {
583         colCount++;
584         --maxCols;
585         columns.add("ALL");
586       } else {
587         colCount += entry.getValue().size();
588         if (maxCols <= 0) {
589           continue;
590         } 
591         for (byte [] column : entry.getValue()) {
592           if (--maxCols <= 0) {
593             continue;
594           }
595           columns.add(Bytes.toStringBinary(column));
596         }
597       } 
598     }       
599     map.put("totalColumns", colCount);
600     if (this.filter != null) {
601       map.put("filter", this.filter.toString());
602     }
603     // add the id if set
604     if (getId() != null) {
605       map.put("id", getId());
606     }
607     return map;
608   }

610   //Writable
611   public void readFields(final DataInput in)
612   throws IOException {
613     int version = in.readByte();
614     if (version > (int)SCAN_VERSION) {
615       throw new IOException("version not supported");
616     }
617     this.startRow = Bytes.readByteArray(in);
618     this.stopRow = Bytes.readByteArray(in);
619     this.maxVersions = in.readInt();
620     this.batch = in.readInt();
621     this.caching = in.readInt();
622     this.cacheBlocks = in.readBoolean();
623     if(in.readBoolean()) {
624       this.filter = Classes.createWritableForName(
625         Bytes.toString(Bytes.readByteArray(in)));
626       this.filter.readFields(in);
627     }
628     this.tr = new TimeRange();
629     tr.readFields(in);
630     int numFamilies = in.readInt();
631     this.familyMap =
632       new TreeMap<byte [], NavigableSet<byte []>>(Bytes.BYTES_COMPARATOR);
633     for(int i=0; i<numFamilies; i++) {
634       byte [] family = Bytes.readByteArray(in);
635       int numColumns = in.readInt();
636       TreeSet<byte []> set = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
637       for(int j=0; j<numColumns; j++) {
638         byte [] qualifier = Bytes.readByteArray(in);
639         set.add(qualifier);
640       }
641       this.familyMap.put(family, set);
642     }
643 
644     if (version > 1) {
645       readAttributes(in);
646     }
647   }

649   public void write(final DataOutput out)
650   throws IOException {
651     out.writeByte(SCAN_VERSION);
652     Bytes.writeByteArray(out, this.startRow);
653     Bytes.writeByteArray(out, this.stopRow);
654     out.writeInt(this.maxVersions);
655     out.writeInt(this.batch);
656     out.writeInt(this.caching);
657     out.writeBoolean(this.cacheBlocks);
658     if(this.filter == null) {
659       out.writeBoolean(false);
660     } else {
661       out.writeBoolean(true);
662       Bytes.writeByteArray(out, Bytes.toBytes(filter.getClass().getName()));
663       filter.write(out);
664     }
665     tr.write(out);
666     out.writeInt(familyMap.size());
667     for(Map.Entry<byte [], NavigableSet<byte []>> entry : familyMap.entrySet()) {
668       Bytes.writeByteArray(out, entry.getKey());
669       NavigableSet<byte []> columnSet = entry.getValue();
670       if(columnSet != null){
671         out.writeInt(columnSet.size());
672         for(byte [] qualifier : columnSet) {
673           Bytes.writeByteArray(out, qualifier);
674         }
675       } else {
676         out.writeInt(0);
677       }
678     }
679     writeAttributes(out);
680   }
